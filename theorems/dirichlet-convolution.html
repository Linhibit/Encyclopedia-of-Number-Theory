function getDivisors(n) {
    const divisors = [];
    const sqrt = Math.sqrt(n);
    
    for (let i = 1; i <= sqrt; i++) {
        if (n % i === 0) {
            divisors.push(i);
            if (i !== n / i) {
                divisors.push(n / i);
            }
        }
    }
    
    return divisors.sort((a, b) => a - b);
}

function dirichletConvolution(f, g, n) {
    const divisors = getDivisors(n);
    let result = 0;
    
    for (const d of divisors) {
        result += f(d) * g(n / d);
    }
    
    return result;
}

function eulerPhi(n) {
    let result = n;
    let temp = n;
    
    for (let p = 2; p * p <= temp; p++) {
        if (temp % p === 0) {
            while (temp % p === 0) {
                temp /= p;
            }
            result -= result / p;
        }
    }
    
    if (temp > 1) {
        result -= result / temp;
    }
    
    return result;
}
