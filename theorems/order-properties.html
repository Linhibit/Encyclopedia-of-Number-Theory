function eulerPhi(n) {
    let result = n;
    let temp = n;
    
    for (let p = 2; p * p <= temp; p++) {
        if (temp % p === 0) {
            while (temp % p === 0) {
                temp /= p;

            result -= result / p;
        }

    if (temp > 1) {
        result -= result / temp;
    }
    
    return result;
}

function getDivisors(num) {
    const divisors = [];

    for (let i = 1; i <= sqrt; i++) {

            divisors.push(i);

                divisors.push(num / i);
            }
        }
    }
    
    return divisors;
}

function modPow(base, exp, mod) {
